#include <cstdio>
using namespace std;
int n;
int a[32][32];
void dfs(int x, int y)
{
    if(x >= 0 && x <= n + 1 && y >= 0 && y <= n + 1)
    {
        if(a[x][y] == 1 || a[x][y] == 3) 
        	return ;
        else
        {
        	a[x][y] = 3;
            dfs(x + 1, y); 
            dfs(x - 1, y);
            dfs(x, y + 1); 
            dfs(x, y - 1);
        }
    }
}
int main()
{
    scanf("%d", &n);
    for(int i = 1; i <= n; ++ i)
    	for(int j = 1; j <= n; ++ j)
    		scanf("%d", &a[i][j]);

    		if(a[i][j] == 3)

    {
    	for(int j = 1; j <= n; ++ j)

    	printf("\n")
    }
    return 0;
}

	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)

			cin>>ch;
			if(ch=='1')
				map[i][j]=true;
		}

	{
		int x,y;

void bfs(int x,int y)
{
	int dx[4]={0,0,-1,1};
    int dy[4]={1,-1,0,0};
    int ans,f,r,newx,newy;
    ans=f=r=1;
    q[f].x=x;
    q[f].y=y;
    memset(flag,false,sizeof(flag));
    flag[x][y]=true;
    while(f<=r)
    {
    	for(int i=0;i<4;i++)
    	{
    		newx=q[f].x+dx[i];
    		newy=q[f].y+dy[i];
    		if(newx>0 && newx<=n && newy>0 && newy<=n && !flag[newx][newy])

    			{
    				r++;
    				ans++;
    				flag[newx][newy]=true;
    				q[r].x=newx;
    				q[r].y=newy;
				}
		}
		f++;
	}
	cout<<ans<<endl;
 }

#include<cstdio>
#include<cstring>
int n,m,x,y;
int ans[100002],f[1002][1002];
char s[1002][1002];
void dfs(int r,int c,int z,int lll)
{
    if (r<0 || r>=n || c<0 || c>=n || f[r][c]!=-1 || s[r][c]-'0'!=z)
        return;
    f[r][c]=lll;
    ans[lll]++;
    dfs(r-1,c,!z,lll);
    dfs(r+1,c,!z,lll);
    dfs(r,c-1,!z,lll);
    dfs(r,c+1,!z,lll);
}
int main()
{
    scanf("%d%d",&n,&m);
    for (int i=0;i<n;i++)
    	scanf("%s",s[i]);
    memset(f,-1,sizeof(f));
    for (int i=0;i<m;i++)
    {
        scanf("%d%d",&x,&y);
        x--;
        y--;
        if (f[x][y]==-1)
            dfs(x,y,s[x][y]-'0',i);
        else 
            ans[i]=ans[f[x][y]];
    }
    for (int i=0;i<m;i++)
    	printf("%d\n",ans[i]);
    return 0;
}
